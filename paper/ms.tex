\begin{eqnarray}
k(\tau) &=&
    \frac{a}{1+b+m}\,
    \exp\left(-\frac{2\pi f \tau }{P\left(1+b+m\right)}\right)
    \left[ b+m
    +    \cos \left(\frac{2 \pi \tau}{P}\right)
    + f\,\sin \left(\frac{2 \pi \tau}{P}\right)
    \right]
\end{eqnarray}
where
\begin{eqnarray}
m &=& \sqrt{1+f^2} \quad.
\end{eqnarray}
This has the properties
\begin{eqnarray}
k(\tau) \ge 0
\,,\quad
k(0) = a
\,,\quad\mathrm{and}\quad
\left.\frac{\dd k(\tau)}{\dd\tau}\right|_{\tau=0} = 0
\quad.
\end{eqnarray}
